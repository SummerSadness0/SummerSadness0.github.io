## 快速排序
**递归**

``` java
    public static void quicksort(int[] arr, int low, int high){
        if(arr == null || arr.length == 0)
            return;
        if(low >= high)
            return;
        int target = arr[high];
        int left = low-1;
        int right = high;
        int index = low;
        while(index < right){
            if(arr[index] < target){
                swap(arr, ++left,index++);
            }else if(arr[index] > target){
                swap(arr, --right, index);
            }else{
                index++;
            }
        }
        swap(arr, index, high);
        quicksort(arr, low, index-1);
        quicksort(arr, index+1, high);
    }
```

 **非递归** 

    得到分隔点位置后，待排序左边部分的左边界和右边界入栈,
    待排序右边部分的左边界和右边界入栈,不断从栈中取出两个边界处理，
    如果还有未排序部分，再入栈，出栈入栈直到栈空
``` java
public class testQuickSort {
    public static void Quick_Sort(int[] nums){
        quickSort(nums, 0, nums.length-1);
    }
    public static void quickSort(int[] nums, int low, int high){
        int par = partition(nums, low, high);
        Stack<Integer> s = new Stack<>();
        if(par > low+1){
            s.push(low);
            s.push(par-1);
        }
        if(par < high-1){
            s.push(par+1);
            s.push(high);
        }
        while(!s.isEmpty()){
            int end = s.pop();
            int start = s.pop();
            int par2 = partition(nums, start, end);
            if(par2 > start+1){
                s.push(start);
                s.push(par2-1);
            }
            if(par2 < end-1){
                s.push(par2+1);
                s.push(end);
            }
        }
    }
    public static int partition(int[] nums, int start, int end){
        int temp = nums[end];
        int low = start-1;
        int high = end;
        int index = start;
        while(index < high){
            if(nums[index] < temp){
                swap(nums, ++low, index++);
            }else if(nums[index] > temp){
                swap(nums, --high, index);
            }else{
                index++;
            }
        }
        swap(nums, high, end);
        return high;
    }
}
```

## 希尔排序

```java
public static void shellSort(int[] arr){
    if(arr == null || arr.length == 0)
        return;
    int n = arr.length;
    for(int gap = n/2; gap > 0; gap /= 2){
        for(int i = gap; i < n; i += gap){
            int temp = arr[i];
            int j = i;
            while(j > 0 && arr[j-gap]>temp){
                arr[j] = arr[j-gap];
                j -= gap;
            }
            arr[j] = temp;
        }
    }
}
```

## 选择排序
```java
public static void selectSort(int[] arr){
    if(arr == null || arr.length == 0)
        return;
    for(int i = 0; i < arr.length; i++){
        int minIndex = i;
        for(int j = i; j < arr.length; j++){
            if(arr[j] < arr[minIndex])
                minIndex = j;
        }
        if(minIndex != i)
            swap(arr, minIndex, i);
    }
}
```
## 归并排序
```java
public static void mergeSort(int[] arr, int low, int high){
    if(low >= high)
        return;
    int mid = low + (high-low)/2;
    mergeSort(arr, low, mid);
    mergeSort(arr, mid+1, high);
    merge(arr, low, high);
}
public static void merge(int[] arr, int low, int high){
    int[] temp = new int[high-low+1];
    int mid = low + (high - low)/2;
    int k = 0;
    int i = low;
    int j = mid+1;
    while(i <= mid && j <= high){
        if(arr[i] < arr[j])
            temp[k++] = arr[i++];
        else
            temp[k++] = arr[j++];
    }
    while(i <= mid)
        temp[k++] = arr[i++];
    while(j <= high)
        temp[k++] = arr[j++];
    for(int t = low; t <= high; t++)
        arr[t] = temp[t-low];
}
```

public int findStrPos(String sourceStr, String patternStr){
    int i = 0;
    int j = 0;
    char[] s = sourceStr.toCharArray();
    char[] p = patternStr.toCharArray();
    int sLen = s.length();
    int pLen = p.length();
    int[] next = getNextArray(patternStr);
    while(i < sLen && j < pLen){
        if(j == -1 || s[i] == p[j]){
            i++;
            j++;
        }else{
            j = next[j];
        }
    }
    if(j == pLen)
        return i - j;
    return -1;
}
public int[] getNextArray(char[] p){
    int pLen = p.length;
    int[] next = new int[pLen];
    int k = -1;
    int j = 0;
    next[0] = -1;
    while(j < pLen - 1){
        if(k == -1 || p[j] == p[k]){
            k++;
            j++;
            next[j] = k;
        }else{
            k = next[k];
        }
    }
    return next;
}
